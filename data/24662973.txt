Item(by='fxtentacle', descendants=None, kids=None, score=None, time=1601649027, title=None, item_type='comment', url=None, parent=24661903, text='Make sure you use HTTP 1.1 or newer and keep-alive. That way, you only have the SSL negotiation overhead on the first HTTPS request, as for all following requests, the connection will be re-used. And then, 100ms for the first request and 15ms for each following request sounds reasonable to me.')

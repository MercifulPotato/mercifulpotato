Item(by='thedufer', descendants=None, kids=None, score=None, time=1611546962, title=None, item_type='comment', url=None, parent=25898035, text='JSON has no floating point numbers.  It just has decimal numbers, which can be arbitrarily large and arbitrarily precise.  If you deserialize them into floats and then are upset that they aren&#x27;t exactly right, that&#x27;s on you, not JSON.  Deserialize them into some bigdecimal type if this is a problem for your domain.')

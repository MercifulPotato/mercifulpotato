Item(by='scarface74', descendants=None, kids=None, score=None, time=1606974305, title=None, item_type='comment', url=None, parent=25271975, text='The Etag is not a reliable hash of the file contents. It will be different if the file was uploaded in multiple parts like using the CLI than if it was moved as one operation like copying from one bucket to another. You can tell the difference because the Etag will have a “-“ in it if it was uploaded as a multipart upload.')

Item(by='AlexITC', descendants=None, kids=None, score=None, time=1606409668, title=None, item_type='comment', url=None, parent=25218567, text='I feel you, while the debugging option is handy for development, it is annoying that you can&#x27;t easily install the extension permanently.<p>The way I do it now is to package the extension as zip (no webext needed) with a custom manifest, which can be installed by setting xpinstall.signatures.required to false in the about:config page.<p>On the manifest side, you just need to include this snippet:<p><pre><code>  &quot;browser_specific_settings&quot;: {\n    &quot;gecko&quot;: {\n      &quot;id&quot;: &quot;example@gmail.com&quot;\n    }\n  }</code></pre>')

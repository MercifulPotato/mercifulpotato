Item(by='layer8', descendants=None, kids=None, score=None, time=1609340153, title=None, item_type='comment', url=None, parent=25582049, text='For me two spaces in fixed-width fonts is too jarring, as the spaces themselves are significantly wider than in proportional fonts (relative to average letter width), and also the period glyph already adds horizontal space.<p>For proportional fonts, on the other hand, two spaces would help the layout engine to distinguish between the end of a sentence and other uses of period-followed-by-space (abbreviations, mostly).')

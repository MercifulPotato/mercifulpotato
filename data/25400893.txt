Item(by='DiogenesKynikos', descendants=None, kids=[25402328, 25402313], score=None, time=1607801887, title=None, item_type='comment', url=None, parent=25400009, text='Energy is not conserved on the scale of the Universe. General Relativity has no energy conservation law (it has a stress-energy tensor conservation law). Two examples of this: expansion of the universe causes the total energy contained in radiation to decrease, and the total amount of dark energy to increase.')

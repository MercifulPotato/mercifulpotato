Item(by='latchkey', descendants=None, kids=None, score=None, time=1603309123, title=None, item_type='comment', url=None, parent=24844029, text='I manage a large datacenter of air cooled fanless (radiator) GPUs housed in cases where the case fans are on the hot side and pulling the air out. Hot air expands, which fills the space (more pressure) and thus prevents cooler air from getting in (you have to push harder to get it in). The faster you can get rid of hot air from the enclosure, the better.')

Item(by='jakelazaroff', descendants=None, kids=None, score=None, time=1606698299, title=None, item_type='comment', url=None, parent=25245724, text='A de jure right to vote often did not translate to actual access to the franchise. States often enacted barriers such as poll taxes and literacy tests, exempting anyone whose grandfather was registered to vote (i.e. not Black people, whose grandparents had been enslaved). Such discriminatory practices were not actually outlawed until the Voting Rights Act was passed a full century later.')

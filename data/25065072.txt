Item(by='Sniffnoy', descendants=None, kids=None, score=None, time=1605143085, title=None, item_type='comment', url=None, parent=25062132, text='That&#x27;s not what the term &quot;directed acyclic graph&quot; means.  &quot;Directed acyclic graph&quot; means that a digraph has no directed cycles, not that it&#x27;s a directed graph with no undirected cycles.  Otherwise the term would just mean a digraph which forms a forest.')

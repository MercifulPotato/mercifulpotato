Item(by='Zak', descendants=None, kids=None, score=None, time=1602181732, title=None, item_type='comment', url=None, parent=24719766, text='Steels with significant amounts of vanadium carbide like S30V should be sharpened with diamond abrasives, as the carbides are harder than most natural stones. They will still take longer to sharpen than less wear-resistant steels, of course, but it should eliminate the difficulty in actually producing a sharp edge.')

Item(by='flohofwoe', descendants=None, kids=None, score=None, time=1604483844, title=None, item_type='comment', url=None, parent=24987686, text='ImGui itself doesn&#x27;t have its own renderer, it just provides a list of simple rendering commands which then must be rendered by your own code (the git repo comes with ready-to-use example renderers using any of the standard 3D-APIs, GL, D3D, Metal, Vulkan, etc...). That way it&#x27;s easy to integrate into existing code bases and portable to all sorts of &quot;exotic&quot; platforms (for instance game consoles - which generally don&#x27;t have &quot;standard&quot; APIs).')

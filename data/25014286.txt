Item(by='JimDabell', descendants=None, kids=None, score=None, time=1604745190, title=None, item_type='comment', url=None, parent=25013989, text='If you didn’t subscribe, don’t unsubscribe.  Set up a filter that auto-reports as spam.  That way you don’t have to see it ever again, but you still give spam filters a fighting chance at stopping <i>other</i> people from seeing it too.  The more you hurt their deliverability, the less effective spamming will become.')

Item(by='Bluestein', descendants=None, kids=None, score=None, time=1608669720, title=None, item_type='comment', url=None, parent=25510986, text='&quot;The Dadabots team, CJ Carr and Zack Zukowski, trained the RNN with two hours of bass improvisation from YouTuber Adam Neely. After some trial and error, Carr and Zukowski limited the dataset to mainly faster licks because the RNN likes fast tempos. That improved the overall sound quality, according to Dadabots, and it means that the AI generates lots of frenetic bass playing.&quot;')

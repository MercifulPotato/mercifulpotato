Item(by='runeks', descendants=None, kids=[25882235, 25881762], score=None, time=1611396559, title=None, item_type='comment', url=None, parent=25881116, text='Bitcoin mining on a block secures not just this block&#x27;s transactions, but the transactions in <i>all previous blocks</i> as well. For each block that is added to the blockchain more transactions are secured by the following blocks. Consequently, these &quot;power usage per transaction&quot;-figures are misleading.')

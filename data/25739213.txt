Item(by='jsherer', descendants=None, kids=None, score=None, time=1610406382, title=None, item_type='comment', url=None, parent=25738173, text='You will not usually have nor do you need line of sight for HF radio. The radio waves are reflected by the ionosphere at various angles and heights, allowing the signal to be received at many different distances at once. Radio waves at the correct frequency hit the ionosphere like stones skipping along the surface of water.')

Item(by='fanf2', descendants=None, kids=None, score=None, time=1609947954, title=None, item_type='comment', url=None, parent=25658456, text='It has both &#x2F; for division with the usual precedence and | for very tight-binding division, for use in fractions. So 1&#x2F;6 cup parses as 1&#x2F;(6 cup) but 1|6 cup parses as (1&#x2F;6) cup.')

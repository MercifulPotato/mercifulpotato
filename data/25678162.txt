Item(by='TD-Linux', descendants=None, kids=[25684319], score=None, time=1610056751, title=None, item_type='comment', url=None, parent=25675047, text='UFI does actually support 1.2MB! This is because many Japanese computers, like the PC-98, spun their 3.5 inch floppies at 360rpm to match the sector layouts of the 5.25 inch one. Many USB floppy drives support this - if you insert a 3.5 inch disk and format it as 1.2MB you&#x27;ll hear the pitch of the drive motor change.')

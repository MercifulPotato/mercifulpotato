Item(by='krastanov', descendants=None, kids=None, score=None, time=1604346851, title=None, item_type='comment', url=None, parent=24971885, text='Your average water droplet acts as a dispersive element (a prism). Dispersive elements can act as temporal Fourier transforms. Lenses act as spatial Fourier transforms. Going further, any mechanical device that exhibits resonance can be claimed to act as a Fourier transform (but that starts sounding meaninglessly vague).')

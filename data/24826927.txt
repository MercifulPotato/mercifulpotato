Item(by='nyanpasu64', descendants=None, kids=None, score=None, time=1603119401, title=None, item_type='comment', url=None, parent=24826518, text='Only apps placed in &#x2F;system&#x2F;priv-app can install other apps in the background. The only way to add an app there is to be Google or a phone manufacturer, or be an end user, root your phone, and modify the filesystem (directly or through Magisk mounts). Users of unrooted phones can&#x27;t install apps with the ability to manage other apps.')

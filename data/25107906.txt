Item(by='vel0city', descendants=None, kids=[25109176], score=None, time=1605499806, title=None, item_type='comment', url=None, parent=25107041, text='You can use OEM licenses on Windows boxes you build yourself. The main difference with an OEM license and a retail license is that OEM licenses do not come with as much support. The idea is the system builder is supposed to provide the first level of support, while retail licenses are supported by Microsoft directly.')

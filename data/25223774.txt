Item(by='abdullahkhalids', descendants=None, kids=[25224076], score=None, time=1606427431, title=None, item_type='comment', url=None, parent=25223469, text='Teen taal means three claps. The idea is that you can divide 16 beat cycle into 4 parts. The first, second and fourth parts are played with the left&#x2F;bass tabla, while the third without.<p>Now, an audience member clapping along is supposed to clap at the start of the first, second and fourth parts, but not at the start of the third part.')

Item(by='mrob', descendants=None, kids=[25808668], score=None, time=1610843831, title=None, item_type='comment', url=None, parent=25806740, text='Beethoven worked as an organist before he began to go deaf, and pipe organs had wider range than modern pianos (as well as playing louder). Despite this he composed very little music for the pipe organ. This makes me think the increased range in his later compositions was just the natural tendency for composers to increase the complexity of their works as they gain experience.')

Item(by='anotherevan', descendants=None, kids=[25182144, 25182225], score=None, time=1606086756, title=None, item_type='comment', url=None, parent=25180662, text='I remember some study with training mice to be fed when pressing a button. They found that if they get a morsel every time they press the button they get bored of it. However if there was some randomisation as to how often they got a treat, they would sit there pressing the button all the time.')

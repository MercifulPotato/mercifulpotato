Item(by='snicker7', descendants=None, kids=[25641497], score=None, time=1609740286, title=None, item_type='comment', url=None, parent=25625707, text='The CPU will load less cache data with linear searches. This is because you will have less cache misses. Less cache misses == less loading from memory. With pointer-heavy data structures, you load more from memory, and most of the stuff you do load is useless.')

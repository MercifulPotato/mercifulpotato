Item(by='MaulingMonkey', descendants=None, kids=[25328733], score=None, time=1607301351, title=None, item_type='comment', url=None, parent=25325818, text='Even programmers who are roughly aware of integral promotion rules may not have connected all the dots to realize the answer is implementation dependent.<p>If sizeof(long) == sizeof(unsigned int), 1L * 2U results in an unsigned int, meaning the result is &quot;false&quot;.<p>If sizeof(long) &gt; sizeof(unsigned int), 1L * 2U result in a <i>signed long</i>, meaning the result is &quot;true&quot;.')

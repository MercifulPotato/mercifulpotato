Item(by='adrian_b', descendants=None, kids=None, score=None, time=1610894895, title=None, item_type='comment', url=None, parent=25810358, text='Originally, &quot;ultramarine&quot; meant something like &quot;imported merchandise&quot;, i.e. something brought from the other side (ultra-) of the sea (-marine).<p>It provided the information that this is a foreign pigment that must be better than the lower-quality pigments produced locally (e.g. azurite).')

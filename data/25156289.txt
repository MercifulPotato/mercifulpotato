Item(by='1vuio0pswjnm7', descendants=None, kids=[25157644], score=None, time=1605833908, title=None, item_type='comment', url=None, parent=25150702, text='Since session cookies are required, here is a simple script for judyrecords searching from command line.  It uses links browser and tmux.<p><pre><code>    #!&#x2F;bin&#x2F;sh\n\n    # usage: 1.sh [query] -- perform search\n    # usage: 1.sh -- process results page 1 to 200\n    # usage: n=5 1.sh -- process results page 5 to 200\n    # usage: n=201 1.sh -- quit\n\n    # start tmux if not already running then detach\n\n    j=https:&#x2F;&#x2F;www.judyrecords.com;\n    case $# in\n    1)\n    tmux set set-remain-on-exit on;\n    tmux neww links;\n    tmux send g $j&#x2F;addSearchJob?search=&quot;$@&quot; c-m;\n    sleep 1.5;\n    tmux send d;\n    tmux capturep -p|sed -n &#x2F;.&#x2F;p;\n    tmux send g $j&#x2F;getSearchJobStatus c-m;\n    sleep 1.5;\n    tmux send d;\n    tmux capture -p|sed -n &#x2F;.&#x2F;p;\n    ;;0)\n    test $n||n=1;while true;do test $n -le 200||break;\n    tmux send g $j&#x2F;getSearchResults?page=$n c-m;\n    sleep 2;\n    tmux send Down Down &#x27;\\&#x27; \n    # small monitor where results page HTML takes 4 spacebar presses to get to bottom; \n    m=0;while true;do test $m -le 3||break;\n    # process results -- e.g., print record URLs;\n    tmux capturep -p|sed -n &quot;&#x2F;href=..record&#x2F;{s|.*record.|$j&#x2F;record&#x2F;|;s&#x2F;\\&quot;&#x2F;&#x2F;;p;}&quot;;\n    tmux send Space;\n    m=$((m+1));done;n=$((n+1));done;\n    tmux killw;\n    esac</code></pre>')

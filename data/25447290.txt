Item(by='jschveibinz', descendants=None, kids=[25447438], score=None, time=1608146873, title=None, item_type='comment', url=None, parent=25446745, text='Convolution is multiplication in the frequency domain.  By convolving with a rectangle, you are multiplying by the Fourier transform of a rectangle in the frequency domain.  In EE terminology, this is a sinc (sin(x)&#x2F;x) low pass filter:  it emphasizes low frequencies and attenuated high frequencies.')

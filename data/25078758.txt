Item(by='jcranmer', descendants=None, kids=[25078856, 25079073, 25078917, 25079104], score=None, time=1605240504, title=None, item_type='comment', url=None, parent=25078601, text='Mail-in ballots are secret ballots. The way they work is that they have two envelopes: you put your ballot in the inner envelope, which has no indication of who is casting the ballot. All of that information is on the outer envelope. When processing the ballot, the information is checked on the outer envelope; once verified, the inner envelope is then handed to another station to be opened and fed into the counting machine while the outer envelope is discarded.')

Item(by='wittyreference', descendants=None, kids=None, score=None, time=1610298012, title=None, item_type='comment', url=None, parent=25714872, text='Actually signing a contract doesn&#x27;t begin and end at a signature. It requires a meeting of the minds, which the signature is meant to represent. A contract without a meeting of the minds - such as a contract presented in a context coercing the counterparty not to have an attorney parse it for them - is invalid, signature or no.')

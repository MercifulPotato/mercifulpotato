Item(by='emptysea', descendants=None, kids=None, score=None, time=1606149625, title=None, item_type='comment', url=None, parent=25188478, text='A nice advantage of having a TextField with a check constraint, instead of a varchar, is that the locks required in Postgres to change the check constraint are less extreme than the ones required to change the length of varchar.<p>With a check constraint, we can use `NOT VALID` with a separate `VALIDATE` call, while a change to the varchar length requires an `ACCESS EXCLUSIVE` lock.')

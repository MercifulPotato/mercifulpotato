Item(by='fanf2', descendants=None, kids=None, score=None, time=1605634036, title=None, item_type='comment', url=None, parent=25123811, text='When git makes packfiles using delta compression, it ignores the history. It roughly sorts blobs for similarity, completely ignoring their filenames or which commits they appear in. This sorting helps to make the packfile delta compression more efficient. The deltas on disk are completely unrelated to the diffs you see from `git show`.')

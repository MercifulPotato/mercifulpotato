Item(by='pbhjpbhj', descendants=None, kids=None, score=None, time=1603195357, title=None, item_type='comment', url=None, parent=24835997, text='You have to use a mark for trade (in UK, EU, at least) to maintain a trademark. So, you can trademark a bunch of stuff but a challenger can point out you didn&#x27;t use the mark and you can lose it - abandonment. A trademark isn&#x27;t a trade mark unless you&#x27;re using it for &#x27;trade&#x27;!<p><i>Not legal advice. Personal opinion, in no way relates to my employment.</i>')

Item(by='karavelov', descendants=None, kids=None, score=None, time=1611531776, title=None, item_type='comment', url=None, parent=25870793, text='Naming fields is a small pain, the PITA is that JSON does not have native encoding for enum&#x2F;variant&#x2F;sum types. So now there are multiple ways to encode them: internally tagged, externally tagged, adjacently tagged, implicit&#x2F;untagged. And JSON schema does not solve that at all: you can write schemas for all of them, but they do not convey the same information in the end languages even if they have all the type capabilities.')

Item(by='ddevault', descendants=None, kids=None, score=None, time=1605208005, title=None, item_type='comment', url=None, parent=25072783, text='As a Stripe user, it has never been smart to use recurring charges, or most of Stripe&#x27;s other advanced, hands-off features. It&#x27;s much, much better to just store payment source IDs and run charges in a cronjob. This gives you the flexibility to leave Stripe more easily when they make changes you don&#x27;t like. Keep your integrations lean, dependencies are a liability.')

Item(by='Bluestein', descendants=None, kids=None, score=None, time=1611427622, title=None, item_type='comment', url=None, parent=25884989, text='&quot;IBM&#x27;s mainframes of the 1960s are based on an interesting technology - Standard Modular System cards or SMS cards. These cards, created in the early years of transistorized computing, each implement a simple circuit on a board about the size of a playing card. You probably think of silicon as the basis for computers, but SMS cards use earlier transistors based on a semiconductor called germanium[1]. Eventually, of course, silicon took over, but in the early 1960s it was germanium not silicon that ran computers.&quot;')

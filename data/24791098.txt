Item(by='ac29', descendants=None, kids=None, score=None, time=1602782435, title=None, item_type='comment', url=None, parent=24790663, text='Leverage does have a specific meaning in investing: it means using debt to amplify returns. This means if the market increases by X%, your investments return &gt;X%, and the same in the negative direction. Having the <i>same</i> returns as the market is not leverage.')

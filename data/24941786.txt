Item(by='bonzini', descendants=None, kids=[24941995, 24942036], score=None, time=1604063232, title=None, item_type='comment', url=None, parent=24941189, text='Never divide by a constant!  Dividing by 5 can be done as a multiplication by 0.3333 hex, i.e. a multiplication by 3 * 17 * 257 followed by a right shift by 16. If you need more precision, just add more factors. Multiplication in turn is done by a hardcoded sequence of shifts and adds.')

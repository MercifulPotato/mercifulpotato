Item(by='StefanKarpinski', descendants=None, kids=None, score=None, time=1610331422, title=None, item_type='comment', url=None, parent=25722532, text='Negative indexing in Python is a dangerous design that hides bugs and causes incorrect programs to return garbage instead of erroring. If you have an incorrect index computation, instead of getting a bounds error, you get different indexing behavior. This makes it dangerously easy to write code that appears to work but computes nonsense.')

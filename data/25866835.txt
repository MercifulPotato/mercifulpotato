Item(by='vgel', descendants=None, kids=[25866956], score=None, time=1611276574, title=None, item_type='comment', url=None, parent=25866609, text='A focus group isn&#x27;t an A&#x2F;B test. An A&#x2F;B test gives you a number per option. A focus group is a conversation with customers where they can actually talk to you and explain their opinions. An A&#x2F;B test could tell you customer click on the darkest blue shade of links the most, but a focus group could tell you that they were clicking on them by mistake.')

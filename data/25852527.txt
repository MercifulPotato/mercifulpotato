Item(by='nine_zeros', descendants=None, kids=None, score=None, time=1611179830, title=None, item_type='comment', url=None, parent=25851569, text='The best way to describe Indian &quot;uncles&quot; and &quot;aunties&quot; is that they are collective nouns for anyone above your own generation who are not your parents or ancestors.<p>They could be related to your parents or friends of your parents or a random person on the street who is older. Kids literally walk into mom-and-pop shops and address the shopkeeper as uncle or aunty.')

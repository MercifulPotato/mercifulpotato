Item(by='nradov', descendants=None, kids=None, score=None, time=1607366331, title=None, item_type='comment', url=None, parent=25334738, text='There is a positive (in the control systems sense) feedback loop with school quality. Whenever test scores drop a little, the most affluent and engaged parents tend to panic. They transfer their children to private school, or move to higher rated school districts. The public school they left behind then suffers further declines in test scores and funding, and the downward spiral continues.')

Item(by='tgsovlerkhgsel', descendants=None, kids=None, score=None, time=1602021852, title=None, item_type='comment', url=None, parent=24702414, text='For (low voltage) lithium-ion batteries, water is the correct extinguishing agent. The amount of metallic lithium contained inside is minimal.<p>Water is easily available, safe, and most importantly, it cools the battery (important if it has multiple cells). Powder could insulate it, causing the other cells to get overheated from the burning one and join the party.')

Item(by='cipherboy', descendants=None, kids=None, score=None, time=1607032216, title=None, item_type='comment', url=None, parent=25295369, text='Most distros have install time dependency resolution, such as yum&#x2F;dnf using libsolv. Repos are a collection of packages alongside metadata tables containing package versions and their dependencies. The tables get cached locally and are used to resolve dependencies prior to downloading any actual RPMs.')

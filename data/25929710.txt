Item(by='namreg', descendants=None, kids=None, score=None, time=1611762695, title=None, item_type='comment', url=None, parent=25929709, text='Go uses an interesting approach to format time. \nInstead of using `%Y-%m-%d` layout you should specify a special layout parameter `Mon Jan 2 15:04:05 MST 2006` the same way as the time or date should be formatted. Sometimes it&#x27;s confusing, especially beginners who have switched from other languages. \nGoDate.io is intended to help convert more familiar date patterns to GO&#x27;s ones.')

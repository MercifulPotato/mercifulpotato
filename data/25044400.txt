Item(by='minetest2048', descendants=None, kids=[25046360, 25044849], score=None, time=1605003974, title=None, item_type='comment', url=None, parent=25044345, text='Note that as the character &#x27;:&#x27; is reserved in NTFS, Proton cannot create its wineprefix in the NTFS steamapps&#x2F;compatdata directory. I&#x27;m not sure whether this will work for games that have native Linux version though')

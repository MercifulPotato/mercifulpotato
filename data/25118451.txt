Item(by='Skunkleton', descendants=None, kids=None, score=None, time=1605569408, title=None, item_type='comment', url=None, parent=25117339, text='Not an expert, but you should end up with more yeast in the end product for a long fermented dough. The yeast population will grow exponentially from whatever initial population was present. You should also end up with the sugars in the flour being more completely broken down into their byproducts (e.g. alcohol, co2).')

Item(by='drallison', descendants=None, kids=None, score=None, time=1609956095, title=None, item_type='comment', url=None, parent=25658733, text='Leap seconds adjust clocks which are run at an atomic standardized rate to meet the astronomical reality and keep UTC in sync.  So far there have be 18 seconds of correction.  Leap seconds are a bad idea. They confuse geophysical data.  The interval of time between two measurements is not the difference between the clock values when the measurements were made.')

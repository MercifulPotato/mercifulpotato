Item(by='cammikebrown', descendants=None, kids=None, score=None, time=1609008629, title=None, item_type='comment', url=None, parent=25545398, text='Let’s say you normally could download 10 kilobytes in a minute. If your computer clock slowed down by half, the kilobytes would download at the same speed, but only 30 seconds would have passed in computer time, so the computer would calculate that you had downloaded 10 kilobytes in 30 seconds, or 20 kilobytes a minute. Really, a minute passes, not 30 seconds, but the computer doesn’t know')

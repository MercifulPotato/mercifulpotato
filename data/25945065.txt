Item(by='ddevault', descendants=None, kids=[25945517], score=None, time=1611854057, title=None, item_type='comment', url=None, parent=25944757, text='Open source is not a business model. You can make money in open source, but not by accident. You have to find a viable business which incorporates open source. One common approach is SaaS, where the software is open source and you provide value by selling hosted versions, with paid support, competing on the reliability and utility of your infrastructure, expertise of your support staff, and integrity of your business.')

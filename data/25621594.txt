Item(by='conradludgate', descendants=None, kids=[25621611, 25621904], score=None, time=1609679749, title=None, item_type='comment', url=None, parent=25621554, text='Comparing floats by equality is a dangerous pattern. It&#x27;s easy for small precision errors to occur. You should instead check that they are close enough to each other, using an epsilon that you find appropriate, perhaps 1e-10. `(x - y).abs() &lt; epsilon` should do the trick')

Item(by='texse', descendants=None, kids=[25566511], score=None, time=1609211305, title=None, item_type='comment', url=None, parent=25566011, text='Most recent tape standards have multiple &quot;bands&quot; and &quot;wraps&quot; placed in parallel. The head reads only one wrap at a time, so it takes many passes to read the whole tape. For example an LTO-8 tape has 52 wraps each within 4 bands, requiring 208 passes to read completely.')

Item(by='andreygrehov', descendants=None, kids=None, score=None, time=1611150053, title=None, item_type='comment', url=None, parent=25845434, text='Panics are rare events that very few functions should ever need to think about. If the library truly cannot set itself up, it might be reasonable to panic (which is why panics are usually in the `main` package only). Once all the invariants have been checked, there is no reason to panic anytime after.')

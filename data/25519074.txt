Item(by='GordonS', descendants=None, kids=[25519260], score=None, time=1608739966, title=None, item_type='comment', url=None, parent=25518831, text='A point of interest, not mentioned in the article, is that loperamide is actually an opioid. All opioids cause some degree of constipation, hence why loperamide helps with diarrhea. Loperamide is unusual, in that it doesn&#x27;t cross the blood-brain barrier, hence why taking it doesn&#x27;t get you high.')

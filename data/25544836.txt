Item(by='buckminster', descendants=None, kids=None, score=None, time=1609003654, title=None, item_type='comment', url=None, parent=25544200, text='A lot of printers, by default, function like an ancient teletype.  They interpret incoming data as something like ASCII and print out the text.  You need to send the correct escape sequence (i.e. a sequence of bytes starting with 0x1B) to switch into a more modern mode.')

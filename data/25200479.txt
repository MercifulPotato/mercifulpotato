Item(by='nitrogen', descendants=None, kids=[25201194], score=None, time=1606238027, title=None, item_type='comment', url=None, parent=25199804, text='Giving each customer a guaranteed bandwidth and a burst bandwidth should work.  When the network is free you can use up to your burst speed, but the network is guaranteed to be large enough to support your minimum speed.  During congestion, all packets beyond your minimum speed are at a lower priority and will be dropped as needed.')

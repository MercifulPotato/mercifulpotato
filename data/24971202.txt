Item(by='gamblor956', descendants=None, kids=[24971405], score=None, time=1604341051, title=None, item_type='comment', url=None, parent=24969413, text='The DMCA does not require the technical measure to be highly effective, or even that it be effective at all.<p>It simply requires that a technical measure of some sort exists.<p>Thus, it doesn&#x27;t matter that it&#x27;s hanging the key next to the door; it is sufficient for the purposes of the DMCA that there is a closed door.')

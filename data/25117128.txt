Item(by='klodolph', descendants=None, kids=[25117635], score=None, time=1605561705, title=None, item_type='comment', url=None, parent=25117109, text='MIDI is a very poor substitute for sheet music, and vice versa. Think of MIDI as a performance and sheet music as the text being performed.<p>If you want to analyze the music or perform it, you want the sheet music. If you want a computer to perform the music, you want the MIDI.')

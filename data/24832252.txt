Item(by='freeone3000', descendants=None, kids=[24833887], score=None, time=1603150210, title=None, item_type='comment', url=None, parent=24830107, text='Money laundering does not require the actual source of the money be illegal. You can fall foul of 18 USC 1956 through a.2.B.ii route, where merely deliberately concealing the source or destination is punishable by a fine for the entire amount concealed. a.2.C provides for 2 years in prison if you didn&#x27;t report when you had to. This is not a questionable interpretation, this is the law as written -- mixing is illegal on its face, no matter the source or destination.')

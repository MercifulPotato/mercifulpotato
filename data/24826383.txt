Item(by='Zigurd', descendants=None, kids=None, score=None, time=1603115869, title=None, item_type='comment', url=None, parent=24825577, text='Patterns taken from manufacturing do not always translate well to software. Kanban is a notable exception. The &quot;pull&quot; model of kanban fits well with the Agile approach to late resource allocation to tasks. Kanban, while it was born in the repeating process world of manufacturing, translates well to the non-repeating, yet iterative nature of software projects.')

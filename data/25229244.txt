Item(by='8bitsrule', descendants=None, kids=None, score=None, time=1606489724, title=None, item_type='comment', url=None, parent=25225473, text='An idea for lottery payouts that are fun for a lot more people. Set a top-prize limit - say $50M. Now suppose the jackpot has risen to $200M. When someone claims the top prize, pay them the $50M. Then select 150 more winning numbers, randomly picked from the numbers sold, and pay each $1M. Or pay out 1500 $100,000 winners.')

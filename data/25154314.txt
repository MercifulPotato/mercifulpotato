Item(by='jefftk', descendants=None, kids=None, score=None, time=1605819734, title=None, item_type='comment', url=None, parent=25154172, text='This is already built into the browser.  Your browser sends &quot;Accept-Encoding&quot; listing what decoders it has (ex: &quot;gzip, deflate, br&quot;), and then the server may encode the body with any of them.  If it does, it will send &quot;Content-Encoding: gzip&quot; or similar to indicate which it chose.')

Item(by='zerocrates', descendants=None, kids=[24660339], score=None, time=1601609133, title=None, item_type='comment', url=None, parent=24657739, text='In general, bitmap fonts are not copyrightable in the United States. Scalable fonts like TrueType and so on are protected as computer programs, but <i>typefaces</i> are not copyrightable and bitmap fonts are just direct representations of the typeface.<p>Other countries have different treatment of typefaces, though.')

Item(by='singlow', descendants=None, kids=None, score=None, time=1602689820, title=None, item_type='comment', url=None, parent=24777585, text='Preferential Voting is a system by which you can rank your preferences rather than choosing a single candidate. You might range Green party first, Democrat second. There are various methods of tallying the votes, but it usually means that you can support the less popular candidate without giving up your weight in the choice between two frontrunners.')

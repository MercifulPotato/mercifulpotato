Item(by='privacylawthrow', descendants=None, kids=None, score=None, time=1611669637, title=None, item_type='comment', url=None, parent=25915618, text='The opt out cookie was created by ad networks prior to GDPR when many EU countries allowed for opt in by default. The opt out cookie was the tool to allow users to opt out. It still has value today as it allows an ad network to remember a user&#x27;s choice not to be tracked.<p>The opt out cookie is set by the advertiser, not the publisher, and the contents of the cookie have generic text like &quot;OPT OUT&quot;.')

Item(by='aden1ne', descendants=None, kids=[25847109], score=None, time=1611153965, title=None, item_type='comment', url=None, parent=25844714, text='I&#x27;ve always found the English language to simply be lacking here. In for instance Dutch there is a very clear distinction between half-open and closed intervals: &quot;x tot y&quot; is always half-open, &quot;x tot en met y&quot; is always closed. Whereas in English, &quot;x until y&quot; can mean both things, with all the resulting misunderstandings.')

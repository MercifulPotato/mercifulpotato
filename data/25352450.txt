Item(by='dralley', descendants=None, kids=None, score=None, time=1607467090, title=None, item_type='comment', url=None, parent=25351196, text='That is not how the GPL works.  The GPL only entitles you to the source code of software for which you have been provided binaries.  If the software has not been provided to you in binary form, you have no claim to the source.<p>This is why the cloud providers can get away with custom in-house patches to the Linux kernel.')

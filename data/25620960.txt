Item(by='uselpa', descendants=None, kids=None, score=None, time=1609670210, title=None, item_type='comment', url=None, parent=25619414, text='Each project can have more or less the same metrics, but different priorities. In a backup tool for example, integrity &gt; restore speed &gt; backup speed (unless the backup speed is so slow that you cannot backup). In a project that can afford to loose data, speed may be more important than integrity.')

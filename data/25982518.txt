Item(by='danbolt', descendants=None, kids=None, score=None, time=1612126759, title=None, item_type='comment', url=None, parent=25973593, text='The Xbox Adaptive controllers are a remarkable device as far as OEM gamepads go. Every input can be specialized and hotswapped with switches connected via headphone jacks. Existing DirectInput USB controllers can also be plugged in to the device as well and used in tandem.<p>The customizability and hackability is through the roof, and the cherry on top is that they work on existing computers and Xbox consoles.')

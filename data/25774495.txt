Item(by='ahartmetz', descendants=None, kids=[25774618], score=None, time=1610618696, title=None, item_type='comment', url=None, parent=25774237, text='Electron beams interfere in ways that light beams don&#x27;t, though. If they cross in vacuum, they deflect each other due to the electric charge carried by electrons. Light beams can only deflect each other in media with non-linear response to electromagnetic fields, i.e. the response to one beam modifies the response to the other beam.')

Item(by='rusk', descendants=None, kids=None, score=None, time=1611227799, title=None, item_type='comment', url=None, parent=25832512, text='In Irish there is no word for the colour orange, even though there is orange on the flag. The technical term is <i>flan bhui</i> or “flag yellow”. <i>Oraiste</i> is the word for the fruit which is sometimes used to describe the colour.')
